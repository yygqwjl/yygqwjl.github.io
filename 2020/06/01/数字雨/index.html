try :#完成库的导入，如果没有这些库的话会进行异常处理输出except下的print()函数

    import pygame

    from pygame.locals import *

    from random import randint

    import random

    import sys

except :

    print("Load modules error!!")

    exit()

#设置初始值

SCREEN_WIDTH = 1366#游戏界面的宽度

SCREEN_HEIGHT = 768#游戏界面的高度

LOW_SPEED = 5#数字掉落的最低速度

HIGH_SPEED = 10#数字掉落的最高速度

LOW_SIZE = 5#掉落数字的最小大小

HIGH_SIZE = 30#掉落数字的最大大小

#FONT_SIZE = 25#设定数字的大小

FONT_NAME = "arial"#设置字体类型

FREQUENCE = 50#数字掉落的频率

times = 0


def randomcolor() :#获取掉落的数字颜色 第一行的return是掉落的数字颜色是随机的
                   #第二行的return 是设定了绿色，达到黑客帝国那样的视觉效果，你们可以自己设定

    #return (randint(0,255),randint(0,255),randint(0,255))

    return(50, 205, 50)

def randomspeed() :#获取随机的掉落速度

    return randint(LOW_SPEED,HIGH_SPEED)

def randomposition() :#获取开始掉落的位置

    return (randint(0,SCREEN_WIDTH),randint(0,SCREEN_HEIGHT))

def randomsize() :

    return randint(LOW_SIZE,HIGH_SIZE)

def randomoname() :

    array=[1,0,2,4]#从array中进行选择掉落的数字，你们可以自己添加

    return random.choice(array)

def randomvalue() :

    array=[1,0,2,4,5]#从array中进行选择掉落的数字，你们可以自己添加

    return random.choice(array)

#class of sprite

class Word(pygame.sprite.Sprite) :#定义一个Word类，继承了pygame.sprite.Sprite的类，就是这个项目的核心部分
                                  #pygame.sprite.Sprite是一个类，是pygame的一个精灵，用来加载动画
    def __init__(self,bornposition) :#__init__自动构造这个类

        pygame.sprite.Sprite.__init__(self)#完成初始化，父类的构造函数

        self.value = randomvalue()#随机获取数字

        self.font = pygame.font.SysFont(FONT_NAME,randomsize())#设置pygame这个框架的名字和名字的大小

        self.image = self.font.render(str(self.value),True,randomcolor()) #对掉落的数字进行游戏画面处理，变成精灵图像

        self.speed = randomspeed()#构建speed属性

        self.rect = self.image.get_rect()#获取图像的具体大小

        self.rect.topleft = bornposition#确定位置


    def update(self) :#实现字体的滑落

        self.rect = self.rect.move(0,self.speed)#字体下滑

        if self.rect.top > SCREEN_HEIGHT :

            self.kill() #字体滑落超出界面就消失
#init the available modules

pygame.init()

screen = pygame.display.set_mode((SCREEN_WIDTH,SCREEN_HEIGHT)) #设置游戏的界面，高度，宽度

pygame.display.set_caption("HACKER EMPIRE CAPTION RAIN") #设置游戏的标题，你们也可以修改，我这个是黑客帝国数字雨

clock = pygame.time.Clock()

group = pygame.sprite.Group() #对精灵进行存贮

group_count = SCREEN_WIDTH // FONT_SIZE

#窗口主循环，从这里开始，程序启动

while True :

    time = clock.tick(FREQUENCE)#画面更新时间

    for event in pygame.event.get() :

        if event.type == QUIT :# 点击右上角的'X'，终止主循环，退出条件

            pygame.quit() #关闭所有pygame模块

            exit()#终止python程序

    # 更新屏幕
    screen.fill((0,0,0))

    for i in range(0,group_count) :

        group.add(Word((i * FONT_SIZE,-FONT_SIZE)))#实现数字的掉落

    group.update()
    #更新数字的位置
    group.draw(screen)
    #在页面上显示数字
    pygame.display.update()
    #更新画面
